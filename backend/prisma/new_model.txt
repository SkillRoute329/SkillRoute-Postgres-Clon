
// 14. Boletines (Registro de Tiempos Reales)
model BulletinEntry {
  id             Int      @id @default(autoincrement())
  tenantId       Int      @default(1)
  inspectorId    Int
  
  date           DateTime
  serviceNumber  String
  
  location       String   // "Pza. Cerro", "Terminal"
  scheduledTime  String   // "06:15"
  actualTime     String?  // "06:20" (hh:mm)
  delayMinutes   Int      @default(0) // Positivo=Atraso, Negativo=Adelanto
  
  busNumber      String?
  occupancyCount Int?
  occupancyLevel String?  // "Low", "Medium", "High", "Full"
  status         String   @default("Completed")
  
  createdAt      DateTime @default(now())
  updatedAt      DateTime @updatedAt
  
  tenant         Tenant   @relation(fields: [tenantId], references: [id])
  inspector      User     @relation(fields: [inspectorId], references: [id])

  @@index([tenantId, serviceNumber])
  @@index([tenantId, date])
}
